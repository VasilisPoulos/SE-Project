1    package datamodel;

3    import java.util.ArrayList; 
4     
5    public abstract class PatternComposite extends PatternComponent 
6    { 
7     
8        private Integer count;      // so no two patters are created with the same name 
9     
10       /** 
11        * ArrayList of components. 
12        */ 
13       protected ArrayList<PatternComponent> componentsList; 
14    
15       /** 
16        * Calling parent constructor, initializing array. 
17        * 
18        * @param name Name of the composite 
19        */ 
20       public PatternComposite(String name) 
21       { 
22           super(name); 
23           this.count = 0; 
24           this.componentsList = new ArrayList<>(); 
25       } 
26    
27       /** 
28        * Adds a PatternComponent item in ArrayList 
29        * 
30        * @param component new component added to the list 
31        */ 
32       @Override 
33       public void add(PatternComponent component) 
34       { 
35           for (PatternComponent i: this.componentsList) { 
36               if (i.getName().equals(component.getName())) { 
37                   this.count++; 
38                   component.setName(component.getName() + "-" + Integer.toString(this.count)); 
39               } 
40           } 
41           this.componentsList.add(component); 
42       } 
43    
44       /** 
45        * Removes a PatternComponent item from the ArrayList 
46        *     if component List contains the item requested, remove it. 
47        * 
48        * @param patternComponentTitle pattern components name 
49        */ 
50       public void remove(String patternComponentTitle) 
51       { 
52           this.componentsList.removeIf((PatternComponent p) -> p.getName().equals(patternComponentTitle)); 
53    
54       } 
55    
56       public ArrayList<PatternComponent> getComponentsList() { 
57           return componentsList; 
58       } 
59    
60       //TODO: will be implemented in Release 2.0

62    
63       @Override 
64       public PatternComponent getChild() { 
65           return super.getChild(); 
66       } 
67    
68       @Override 
69       public String toString() { 
70           String list=""; 
71           for(int i=0;i<componentsList.size();i++){ 
72               list=componentsList.get(i).toString(); 
73           } 
74           return list; 
75       }
