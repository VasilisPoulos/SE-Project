85       /** 
86        * Clones a pattern prototype (a template) and returns the copy 
87        * 
88        * @param templateName name of the template 
89        * @return Pattern based on template prototype 
90        */ 
91       public Pattern createTemplate(String templateName) 
92       { 
93           return templatesList.get(templateName).clone(); 
94       }
