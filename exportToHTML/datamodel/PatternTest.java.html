15        * the same 
16        */ 
17       @Test 
18       void shouldClone() { 
19    
20           Pattern oldPattern = new Pattern("newPattern"); 
21           PatternPart part = new PatternPart("Lorem ipsum"); 
22           oldPattern.add(part); 
23    
24           /* Test that clone works correctly */ 
25           Pattern newPattern = oldPattern.clone(); 
26           assertEquals(oldPattern.getComponentsList().get(0).getName(), newPattern.getComponentsList().get(0).getName()); 
27           assertNotEquals(oldPattern.getComponentsList().get(0), newPattern.getComponentsList().get(0));

30       }

32       @Test 
33       void shouldMakeDeepCopy() { 
34    
35           Pattern oldPattern = new Pattern("newPattern"); 
36           PatternPart part = new PatternPart("Lorem ipsum"); 
37           oldPattern.add(part); 
38    
39           Pattern newPattern = oldPattern.clone(); 
40    
41           /* Test that clone has different components after changing */ 
42           newPattern.getComponentsList().get(0).setName("dolor sit"); 
43           assertNotEquals(oldPattern.getComponentsList().get(0), newPattern.getComponentsList().get(0)); 
44    
45           /* Test that clone has same size components before adding */ 
46           assertEquals(oldPattern.getComponentsList().size(), newPattern.getComponentsList().size()); 
47    
48           /* Test that clone has different size after adding components */ 
49           newPattern.add(new PatternPart("amet")); 
50           assertNotEquals(oldPattern.getComponentsList().size(), newPattern.getComponentsList().size());
